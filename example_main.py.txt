@bot.command(aliases=["logoff"])
@commands.is_owner()
async def logout(ctx):
    await ctx.send(f"{ctx.message.author} I'm logging off now")
    await bot.logout()

async def logout_error(ctx, error):
    if isinstance(error, commands.CheckFailure):
